grpc+proto小实战

一、proto内容定义

syntax = "proto3"
option go_package = ".;hello"
package = "hello"


// grpc + protobuf

service TestService {

    rpc SayHi (HiRequest) returns (HiResponse){}

}

message HiRequest {
    string name = 1
}

message HiResponse {
    string ret = 1

}

二、转成go文件

注意：加grpc,使用grpc插件
protoc --go_out=plugins=grpc:. say_hi_test.proto

三、grpc服务端

package main

import (
    "net"
    "fmt"
    "google.golang.org/grpc"
    "micro_service_project/grpc_protobuf_test"
    "context"
)

type Server struct {}

func (s *Server) SayHi(ctx context.Context, in *hello.HiRequest) (*hello.HiResponse, error)  {

    rep := hello.HiResponse{Ret:"hi, "+ in.Name}
    return &rep,nil
}



// name  -->   "hi,name"
func main() {

    // 创建grpc服务
    server := grpc.NewServer()

    hello.RegisterTestServiceServer(server,&Server{})

    listen,err := net.Listen("tcp","127.0.0.1:8080")

    fmt.Println("监听8080端口。。。")
    if err != nil {
        fmt.Println("网络错误")
    }

    //http.Serve(listen,nil)   // 不能使用这种方式
    server.Serve(listen)   // grpc服务启动


}

四、grpc客户端

package main

import (
    "google.golang.org/grpc"
    "fmt"
    "micro_service_project/grpc_protobuf_test"
    "context"
)

func main() {

    // WithInsecure：跳过证书的验证
    conn,err := grpc.Dial("127.0.0.1:8080",grpc.WithInsecure())

    if err != nil {
        fmt.Println("连接错误")
    }

    client := hello.NewTestServiceClient(conn)

    rep,err1 := client.SayHi(context.Background(),&hello.HiRequest{Name:"zhiliao"})
    fmt.Println(err1)
    if err1 != nil {

        fmt.Println("返回出错")
    }

    fmt.Println(rep.Ret)
}

报错

google.golang.org\grpc\internal\transport\http_util.go:233:23: cannot use hf (type "vendor/golang.org/x/net/http2/hpack".HeaderField) as type "golang.org/x/net/http2/hpack".HeaderField in argument to d.processHeaderField
google.golang.org\grpc\internal\transport\http_util.go:605:23: cannot use "golang.org/x/net/http2/hpack".NewDecoder(http2InitHeaderTableSize, nil) (type *"golang.org/x/net/http2/hpack".Decoder) as type *"vendor/golang.org/x/net/http2/hpack".Decoder in assignment

解决办法：
1.将我提供的golang.org压缩包放在goroot的src下
2.将goroot下src下的vendor文件夹删除
3.将我提供的vendor压缩包放在gopath下的src下
